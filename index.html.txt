        // –ó–∞–≥—Ä—É–∑–∫–∞ –¥–∞–Ω–Ω—ã—Ö –∏–∑ localStorage
        function loadCounters() {
            const saved = localStorage.getItem('emailCounters');
            const today = new Date().toDateString();
            const savedDate = localStorage.getItem('countersDate');
            
            if (saved && savedDate === today) {
                counters = JSON.parse(saved);
            } else {
                // –°–æ–∑–¥–∞—ë–º –Ω–æ–≤—ã–µ —Å—á—ë—Ç—á–∏–∫–∏ –¥–ª—è —Å–µ–≥–æ–¥–Ω—è—à–Ω–µ–≥–æ –¥–Ω—è
                emailTypes.forEach(type => {
                    counters[type.id] = 0;
                });
                localStorage.setItem('countersDate', today);
                saveCounters();
            }
        }

        // –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –¥–∞–Ω–Ω—ã—Ö –≤ localStorage

        // –î–æ–±–∞–≤–ª–µ–Ω–∏–µ –∫ —Å—á—ë—Ç—á–∏–∫—É
        function addToCounter(typeId, amount) {
            counters[typeId] += amount;
            updateCounterDisplay(typeId);
            saveCounters();
            generateReport();
            showNotification(`–î–æ–±–∞–≤–ª–µ–Ω–æ ${amount} –∫ "${emailTypes.find(t => t.id === typeId).name}"`);

        function subtractFromCounter(typeId, amount) {
            if (counters[typeId] > 0) {
                counters[typeId] = Math.max(0, counters[typeId] - amount);
                updateCounterDisplay(typeId);
                saveCounters();
                generateReport();
                showNotification(`–£–±—Ä–∞–Ω–æ ${amount} –∏–∑ "${emailTypes.find(t => t.id === typeId).name}"`);
            }
        }

                saveCounters();

            if (element) {
                element.textContent = counters[typeId];
                // –ê–Ω–∏–º–∞—Ü–∏—è –∏–∑–º–µ–Ω–µ–Ω–∏—è

                const subtractBtn = element.parentElement.querySelector('.counter-btn-subtract');
                if (subtractBtn) {
                    if (counters[typeId] === 0) {
                        subtractBtn.disabled = true;
                        subtractBtn.style.opacity = '0.5';
                        subtractBtn.style.cursor = 'not-allowed';

                        subtractBtn.disabled = false;
